jQuery Alert Dialogs (Alert, Confirm, & Prompt Replacements)

Basic Examples

Alert 
jAlert('This is a custom alert box', 'Alert Dialog');

Confirm 
jConfirm('Can you confirm this?', 'Confirmation Dialog', function(r) {
	jAlert('Confirmed: ' + r, 'Confirmation Results');
});

Prompt 
jPrompt('Type something:', 'Prefilled value', 'Prompt Dialog', function(r) {
	if( r ) alert('You entered ' + r);
});

Custom Confirm 
jCustomConfirm('Can you confirm this custom dialog?', 'Confirm Header', 'Think about it', 'Maybe Later');

Custom Popup 
var template = 'Define Form Without Submit/Reset Button';
jCustomPopup(template, 'Popup Header', 'Register', 'Cancel',function(r,data){
	console.log(data);
});
